def go(n):
    k = [0, 1]
    for i in range(2, n):
        a = k[i - 2] + k[i - 1]
        k.append(a)
    return k[n - 1]


print(go(10))




def count_trajected(n, allowed: list[bool]):
    k = [0, 1, int(allowed[2])] + [0] * (n - 3)
    for i in range(3, n + 1):
        if allowed[i]:
            k[i] = k[i-1] + k[i-2] + k[i-3]
    return k


x = [True] * 11
print(x)
print(count_trajected(10, x))
