## алгоритм вычисления арифметических выражений в постфиксной нотации 

в коментариях в коде пояснено 
``` python 

from stack import stack_on_list
stack_on_list.clear()


#здесь пояснение!
ex1 = [2,3, '+'] # 2+3
ex2 = [2,7,'+',5, '*']# (2+7)*5
ex3 = [2,7,5,'*','+']#2+7*5
ex4 = [4,3,'-']# 4-3
ex5 = [16, 4, '/']# 16/4


def make_operation(y:int, x:int, operation:str) -> int:
    if operation == '+':
        return y + x
    if operation == '-':
        return x - y
    if operation == '*':
        return y * x
    if operation == '/':
        return x // y


def p_note(notes:list) -> int:
    """
    >>> p_note(ex1)
    5
    >>> p_note(ex2)
    45
    >>> p_note(ex3)
    37
    >>> p_note(ex4)
    1
    >>> p_note(ex5)
    4
    
    """



    for note in notes:
        if str(note) not in '+-*/':
            stack_on_list.push(note)
        else:
            y = stack_on_list.pop()
            x = stack_on_list.pop()
            z = make_operation(y, x, note)
            stack_on_list.push(z)
    return stack_on_list.pop()




if __name__ == "__main__":

    doctest.testmod(verbose=True)
```
